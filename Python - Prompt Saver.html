        prompt_folder = self.tree.insert("", tk.END, text="Prompts", open=True)

        for topic in ["Topic 1", "Topic 2", "Topic 3"]:
            topic_folder = self.tree.insert(prompt_folder, tk.END, text=topic, open=True)

            self.tree.insert(topic_folder, tk.END, values=("Prompt 1"))
